Android使用UrlConnection异步网络编程

Posted on 2015-07-30   |  

Android实现Http请求的方式一般分为

  1. Java接口(java.net)–HttpURLConnection，可以实现简单的基于URL请求、响应功能；
  2. Apache接口(org.appache.http)–HttpClient，使用起来更方面更强大。

QWSDK使用较为原始的HttpURLConnection，以下介绍HttpURLConnection实现异步网络请求的两种方式，整体的流程都是在子线程中处理网络操作(必须)，得到返回结果后交由主线程处理

一、使用线程+Android消息管理机制

二、使用AsyncTask

JAVA代码如下：

public class MainActivity extends Activity {
    Button visitWebBtn = null;
    Button downImgBtn = null;
    TextView showTextView = null;
    ImageView showImageView = null;
    String resultStr = "";

    private Handler handle = new Handler(){
        public void handleMessage(Message msg){
            resultStr = msg.getData().getString("data");
            showTextView.setText(resultStr);
        }
    };
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        Log.d("ThreadID", "onCreate: " + Thread.currentThread().getId());
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        initUI();
        visitWebBtn.setOnClickListener(new View.OnClickListener() {

            @Override
            public void onClick(View v) {
                Log.d("ThreadID", "visitWebBtn onClick: " + Thread.currentThread().getId());
                showImageView.setVisibility(View.GONE);
                showTextView.setVisibility(View.VISIBLE);
                Thread visitBaiduThread = new Thread(new VisitWebRunnable());
                visitBaiduThread.start();
                /*以下为通过等待子线程结束后获取网络请求返回的值，若用这种方式则需要在VisitWebRunnable的run中将resultStr赋值
                  不推荐用这种方式，因为这样主线程需要等待子线程结果，结果出来之前会卡住
                try {
                    Log.d("ThreadID","brfore join: "+Thread.currentThread().getId());
                    visitBaiduThread.join();//等待更新resultStr参数
                    Log.d("ThreadID", "after join: " + Thread.currentThread().getId());
                    if(!resultStr.equals("")){
                        showTextView.setText(resultStr);
                    }
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
                */
            }
        });
        downImgBtn.setOnClickListener(new View.OnClickListener() {

            @Override
            public void onClick(View v) {
                Log.d("ThreadID", "downImgBtn onClick: " + Thread.currentThread().getId());
                showImageView.setVisibility(View.VISIBLE);
                showTextView.setVisibility(View.GONE);
                String imgUrl = "http://hiphotos.baidu.com/%CD%ED%C9%CF%B3%AA%B8%E8%CF%C2%D3%EA/pic/item/e3afd71367a912155baf5330.jpg";
                new DownImgAsyncTask().execute(imgUrl);
            }
        });
    }

    public void initUI(){
        showTextView = (TextView)findViewById(R.id.textview_show);
        showImageView = (ImageView)findViewById(R.id.imagview_show);
        downImgBtn = (Button)findViewById(R.id.btn_download_img);
        visitWebBtn = (Button)findViewById(R.id.btn_visit_web);
    }

    private String getURLResponse(String urlString){
        HttpURLConnection conn = null;
        InputStream is = null;
        String resultData = "";
        try {
            Log.d("ThreadID", "getURLResponse : " + Thread.currentThread().getId());
            URL url = new URL(urlString); //URL对象
            conn = (HttpURLConnection)url.openConnection(); //使用URL打开一个链接
            conn.setDoInput(true); //允许输入流，即允许下载
            conn.setDoOutput(true); //允许输出流，即允许上传
            conn.setUseCaches(false); //不使用缓冲
            conn.setRequestMethod("GET"); //使用get请求
            is = conn.getInputStream();   //获取输入流，此时才真正建立链接
            Log.d("ThreadID", "getURLResponse getInputStream: " + Thread.currentThread().getId());
            InputStreamReader isr = new InputStreamReader(is);
            BufferedReader bufferReader = new BufferedReader(isr);
            String inputLine  = "";
            while((inputLine = bufferReader.readLine()) != null) {
                resultData += inputLine + "\n";
            }
        }catch (IOException e) {
            e.printStackTrace();
        }finally{
            if(is != null){
                try {
                    is.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            if(conn != null){
                conn.disconnect();
            }
        }

        return resultData;
    }

    private Bitmap getImageBitmap(String url){
        URL imgUrl;
        Bitmap bitmap = null;
        try {
            imgUrl = new URL(url);
            HttpURLConnection conn = (HttpURLConnection)imgUrl.openConnection();
            conn.setDoInput(true);
            conn.connect();
            InputStream is = conn.getInputStream();
            Log.d("ThreadID", "doInbackground getInputStream: " + Thread.currentThread().getId());
            bitmap = BitmapFactory.decodeStream(is);
            is.close();
        }catch(IOException e){
            e.printStackTrace();
        }
        return bitmap;
    }

    class VisitWebRunnable implements Runnable{

        @Override
        public void run() {
            Log.d("ThreadID"," VisitWebRunnable run: "+Thread.currentThread().getId());
            String data = getURLResponse("http://www.baidu.com/");
            Message message = new Message();
            Bundle bundle = new Bundle();
            bundle.putString("data",data);
            message.setData(bundle);
            handle.sendMessage(message);
            //resultStr = data;
        }

    }

    class DownImgAsyncTask extends AsyncTask<String, Void, Bitmap>{

        @Override
        protected void onPreExecute() {
            super.onPreExecute();
            showImageView.setImageBitmap(null);
            Log.d("ThreadID", "onPreExecute: " + Thread.currentThread().getId());

        }

        @Override
        protected Bitmap doInBackground(String... params) {
            Bitmap b = getImageBitmap(params[0]);
            Log.d("ThreadID","doInBackground: "+Thread.currentThread().getId());
            return b;
        }

        @Override
        protected void onPostExecute(Bitmap result) {
            Log.d("ThreadID","onPostExecute: "+Thread.currentThread().getId());
            super.onPostExecute(result);
            if(result!=null){
                showImageView.setImageBitmap(result);
            }
        }
    }
}
